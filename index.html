using System; using System.Threading.Tasks; using System.Windows.Forms; using System.Net.Http; using System.Text; namespace GitHubTextEditor { public partial class Form1 : Form { public Form1() { InitializeComponent(); } private async void buttonLoad_ClickAsync(object sender, EventArgs e) { string fileUrl = "https://api.github.com/repos/nabiAcram/1.txt/contents/1.txt"; // تغییر به URL API string fileContent = await LoadFileFromGitHub(fileUrl); if (!string.IsNullOrEmpty(fileContent)) { richTextBoxTxtEditor.Text = fileContent; } } private async Task LoadFileFromGitHub(string fileUrl) { using (HttpClient client = new HttpClient()) { client.DefaultRequestHeaders.Add("User-Agent", "GitHubTextEditor"); HttpResponseMessage response = await client.GetAsync(fileUrl); if (response.IsSuccessStatusCode) { string jsonResponse = await response.Content.ReadAsStringAsync(); // استخراج داده base64 از JSON var fileInfoJson = Newtonsoft.Json.Linq.JObject.Parse(jsonResponse); string base64Content = fileInfoJson["content"].ToString(); // تبدیل از base64 به متن byte[] decodedBytes = Convert.FromBase64String(base64Content); string fileContent = Encoding.UTF8.GetString(decodedBytes); return fileContent; } else { MessageBox.Show("Error loading file from GitHub"); return string.Empty; } } } private async void buttonSave_ClickAsync(object sender, EventArgs e) { var result = MessageBox.Show("آیا می‌خواهید تغییرات را آپلود کنید؟", "تایید آپلود", MessageBoxButtons.YesNo, MessageBoxIcon.Question); if (result == DialogResult.Yes) { string fileContent = richTextBoxTxtEditor.Text; string token = "your-github-token"; // توکن گیت‌هاب خود را اینجا وارد کنید string fileUrl = "https://api.github.com/repos/nabiAcram/1.txt/contents/1.txt"; // تغییر به URL API bool success = await UploadFileToGitHub(fileUrl, fileContent, token); if (success) { MessageBox.Show("فایل با موفقیت ویرایش و آپلود شد"); } else { MessageBox.Show("Error uploading file to GitHub."); } } if (result == DialogResult.No) { MessageBox.Show("ویرایش کنسل شد"); } } private async Task UploadFileToGitHub(string fileUrl, string fileContent, string token) { using (HttpClient client = new HttpClient()) { client.DefaultRequestHeaders.Add("Authorization", "token " + token); client.DefaultRequestHeaders.Add("User-Agent", "GitHubTextEditor"); HttpResponseMessage response = await client.GetAsync(fileUrl); if (!response.IsSuccessStatusCode) { string errorMessage = await response.Content.ReadAsStringAsync(); MessageBox.Show("Error fetching file information: " + errorMessage); return false; } string fileInfo = await response.Content.ReadAsStringAsync(); var fileInfoJson = Newtonsoft.Json.Linq.JObject.Parse(fileInfo); string sha = fileInfoJson["sha"].ToString(); // تبدیل محتوا به بایت با UTF-8 بدون BOM var utf8WithoutBOM = new UTF8Encoding(encoderShouldEmitUTF8Identifier: false); byte[] encodedBytes = utf8WithoutBOM.GetBytes(fileContent); string base64Content = Convert.ToBase64String(encodedBytes); var body = new { message = "Save as UTF-8 without BOM", content = base64Content, sha = sha }; string jsonBody = Newtonsoft.Json.JsonConvert.SerializeObject(body); var content = new StringContent(jsonBody, Encoding.UTF8, "application/json"); response = await client.PutAsync(fileUrl, content); if (!response.IsSuccessStatusCode) { string errorMessage = await response.Content.ReadAsStringAsync(); MessageBox.Show("Error uploading file to GitHub: " + errorMessage); return false; } return true; } } } }
